        function createProductGroupHTML(group) {
            let html = `<div><h2>${group.groupName}</h2>`;
            html += `<table class="productTable"><thead><tr>
                <th>Product Details</th>
                <th>Qty</th>
                <th>Packing</th>
                <th>Weight/PC (KG)</th>
                <th>Ordered Qty</th>

                const qtyType = product.qtyL ? 'L' : (product.qtyML ? 'ML' : (product.qtyG ? 'G' : ''));

        function handleQuantityChange(event) {
            const input = event.target;
            const groupIndex = parseInt(input.dataset.groupIndex);
            const productIndex = parseInt(input.dataset.productIndex);
            const qty = parseInt(input.value) || 0;
            
            const product = productGroups[groupIndex].products[productIndex];
            product.orderedQty = qty;

            const row = input.closest('tr');
            const totalBottles = qty * product.packing;
            const totalBoxes = qty;
            const boxWeight = qty * product.packing * product.weight;

            row.querySelector('.bottles-cell').textContent = totalBottles;
            row.querySelector('.boxes-cell').textContent = totalBoxes;
            row.querySelector('.weight-cell').textContent = boxWeight.toFixed(2);
        }

        function handleCalculateClick() {
            let totalBottles = 0;
            let totalBoxes = 0;
            let totalWeight = 0;

            finalProductTableBody.innerHTML = '';
            
            productGroups.forEach(group => {
                group.products.forEach(product => {
                    const totalB = product.orderedQty * product.packing;
                    const totalX = product.orderedQty;
                    const boxW = product.orderedQty * product.packing * product.weight;
                    
                    if (product.orderedQty > 0) {
                        const row = document.createElement('tr');
                        const qtyType = product.qtyL ? 'L' : (product.qtyML ? 'ML' : 'G');
                        const qtyValue = product.qtyL || product.qtyML || product.qtyG;
                        row.innerHTML = `
                            <td>${product.details} (${qtyValue} ${qtyType})</td>
                            <td>${product.orderedQty}</td>
                            <td>${totalB}</td>
                            <td>${totalX}</td>
                            <td>${boxW.toFixed(2)}</td>
                        `;
                        finalProductTableBody.appendChild(row);
                    }
                    
                    totalBottles += totalB;
                    totalBoxes += totalX;
                    totalWeight += boxW;
                });
            });

            const totalWeightTon = totalWeight / 1000;
            
            productGroupsContainer.style.display = 'none';
            document.querySelector('.action-buttons').style.display = 'none';
            document.getElementById('progressBarContainer').style.display = 'none';

            finalProductListContainer.style.display = 'block';
            summarySection.style.display = 'block';
            
            // FIX: Add the new CSS class to center the header
            header.classList.remove('minimized');
            header.classList.add('summary-view');

            const summaryItems = document.querySelectorAll('.summary-item');

            document.getElementById('grandTotal').textContent = totalWeight.toFixed(2) + ' KG';

            window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' });
        }
